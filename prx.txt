function Start-SleepWithProgress {
    param (
        [Parameter(Mandatory=$true)]
        [int]$Seconds
    )

    $startTime = Get-Date
    $endTime = $startTime.AddSeconds($Seconds)

    while ((Get-Date) -lt $endTime) {
        $elapsedTime = (Get-Date) - $startTime
        $percentComplete = ($elapsedTime.TotalSeconds / $Seconds) * 100
        $secondsRemaining = [math]::Round($Seconds - $elapsedTime.TotalSeconds)

        Write-Progress -Activity "Pausing..." -Status "$percentComplete% Complete" -PercentComplete $percentComplete -SecondsRemaining $secondsRemaining

        Start-Sleep -Seconds 1
    }

    Write-Progress -Activity "Pausing..." -Completed
}



write-output "hello world"



# Example usage:
Start-SleepWithProgress -Seconds 15

exit 0
